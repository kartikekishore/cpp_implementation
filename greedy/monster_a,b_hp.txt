		cin >> arr[i];

		if (arr[i] % (a + b) == 0)
		{kans[i] += a + b;}




		kans[i] = ((kans[i] + a - 1) / a) - 1;



	}
	sort(kans, kans + n);
	int count = 0;
	for ( i = 0; i < n; ++i)
	{
		if (k - kans[i] < 0)
			break;
		count++;
		k -= kans[i];
	}

	cout << count << endl;


}

int mpow(int base, int exp) {
	base %= mod;
	int result = 1;
	while (exp > 0) {
		if (exp & 1) result = (result * base) % mod;
		base = (base * base) % mod;
		exp >>= 1;
	}
	return result;
}
